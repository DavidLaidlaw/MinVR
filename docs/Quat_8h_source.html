256  Quat log() const {
257  if ((x == 0) && (y == 0) && (z == 0)) {
258  if (w > 0) {
259  return Quat(0, 0, 0, ::logf(w));
260  } else if (w < 0) {
261  // Log of a negative number. Multivalued, any number of the form
262  // (PI * v, ln(-q.w))
263  return Quat((float)pi(), 0, 0, ::logf(-w));
264  } else {
265  // log of zero!
266  return Quat((float)nan(), (float)nan(), (float)nan(), (float)nan());
267  }
268  } else {
269  // Partly imaginary.
270  float imagLen = sqrtf(x * x + y * y + z * z);
271  float len = sqrtf(imagLen * imagLen + w * w);
272  float theta = atan2f(imagLen, (float)w);
273  float t = theta / imagLen;
274  return Quat(t * x, t * y, t * z, ::logf(len));
275  }
