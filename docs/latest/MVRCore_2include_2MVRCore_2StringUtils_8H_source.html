170 G3DLite::Array<T> insertIntoArray(G3DLite::Array<T> inputArray, T newElement,
171  int insertBeforeElementNum)
172 {
173  debugAssert(insertBeforeElementNum >= 0);
174  debugAssert(insertBeforeElementNum <= inputArray.size());
175 
176  G3DLite::Array<T> anew;
177 
178  for (int i=0;i<insertBeforeElementNum;i++)
179  anew.append(inputArray[i]);
180 
181  anew.append(newElement);
182 
183  for (int i=insertBeforeElementNum;i<inputArray.size();i++)
184  anew.append(inputArray[i]);
185 
186  debugAssert(anew.size() == (inputArray.size() + 1));
187  return anew;
188 }
