117 template <class KEYTYPE, class VALTYPE>
118 VALTYPE ConfigVal(KEYTYPE keyString, const VALTYPE &defaultVal, bool warn=true) {
119  debugAssertM(ConfigValMap::map.notNull(), "The global config map is NULL!");
120  if (!ConfigValMap::map->containsKey(keyString)) {
121  if (warn) {
122  notFoundWarning(keyString);
123  }
124  return defaultVal;
125  }
126  else {
127  std::string valString = ConfigValMap::map->getValue(keyString);
128  //std::cout << "valString = " << valString << std::endl;
129  VALTYPE val;
130  bool ok = retypeVal(valString, val);
131  if (!ok) {
132  std::string errString = string("ERROR: ConfigVal is unable to retype value.\n\tKey:\t'")
133  + keyString + string("'\n\tValue:\t'") + valString + string("'\n");
134  std::cout << errString;
135  }
136  return val;
137  }
138 }
