792 inline double eps(double a, double b) {
793  // For a and b to be nearly equal, they must have nearly
794  // the same magnitude. This means that we can ignore b
795  // since it either has the same magnitude or the comparison
796  // will fail anyway.
797  (void)b;
798  const double aa = abs(a) + 1.0;
799  if (aa == inf()) {
800  return fuzzyEpsilon;
801  } else {
802  return fuzzyEpsilon * aa;
803  }
